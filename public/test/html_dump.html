create table stock_daily_rates_5y (
`id` bigint(20) NOT NULL AUTO_INCREMENT,
date varchar(15),
open decimal(7,1),
high decimal(9,2),
low decimal(9,2),
close decimal(7,1),
volume bigint(10),
unadjusted_volume bigint(10),
change decimal(9,2),
change_percent decimal(9,2),
vwap decimal(13,4),
change_over_time decimal(31,13),
created_at datetime,
updated_at datetime,
PRIMARY KEY (`id`)
);
